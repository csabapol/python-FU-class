Q3: How did the aerial attacks change over time for the 15 most frequent targeted cities?¶

Count the amount of explosives for individual cities for each year and save it to an Numpy Array

[19]:

unique_years = np.array([i for i in range(df_tar['Mission Date'].dt.year.min(), df_tar['Mission Date'].dt.year.max() + 1)])

city_year = []
# Fuer alle Staedte
for city in cities[0:15]:
    # erstelle temporaere Liste, wobei erster Eintrag Stadtname ist
    year_arr = [city]
    # Fuer Jahre in (1939,...,1945)
    for year in unique_years:
        # Fuege Summe an 'Explosionsstoff' zur temporaeren Liste hinzu
        # year_arr = [Stadtname, Expl. in 1939, 1940, ..., 1945]
        year_arr.append((df_tar.loc[
                            (df_tar['Mission Date'].dt.year == year) &
                            (df_tar['Target City'] == city)
                        ]['High Explosives Weight (Tons)'].sum()))
    # Fuege temporaere Liste zur Gesamtliste hinzu
    city_year.append(year_arr)
city_year = np.array(city_year)

[22]:

df_city_daily_mean = df_city_year.copy()
df_city_daily_mean[1939] = df_city_year[1939]/(30+31+30+31) # WW2 began on 1st September 1939
df_city_daily_mean[[1941,1942,1943]] = df_city_year[[1941,1942,1943]]/365
df_city_daily_mean[[1940,1944]] = df_city_year[[1940,1944]]/366 # 1940 and 1944 were leap-years
df_city_daily_mean[1945] = df_city_year[1945]/(31+28+31+30+8) # WW2 ended on 8th Mai 1945
