import math


def function(df, airplane, years = year_list, base = europe):

    fig = plt.figure(figsize=(20,20))
    years = years.sort()
    for a,b in enumerate(year_list):
        new_df = df.loc[(gdf["year"]==b) & (df["Aircraft Series"]==airplane)]
        new_df["Country"].replace(np.nan, "NaN", inplace=True)
        ax = fig.add_subplot(4,2,a+1)
        europe.plot(ax = ax, facecolor = "lightgray", edgecolor="white")
        ax.set_aspect('equal')
        ax.set_title("Attacks by plane " + str(airplane) +" ("+ str(b)+")")
        new_df.plot(ax = ax, marker = "o", markersize = 7,categorical = True, column = "Country")


    return ax
